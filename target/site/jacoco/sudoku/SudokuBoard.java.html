package sudoku;

import java.util.*;

public class SudokuBoard {
    private int[][] board;

    public SudokuBoard(final int[][] board) { //final -> const
        this.board = board;
        //if (!checkBoard()) {
        //    this.board =  new int[9][9];
        //}
    }

    private boolean checkBoard() {
        for (int row = 0; row < 9; row++) {
            for (int col = 0; col < 9; col++) {
                int num = board[row][col];
                if (num != 0) {
                    if (!checkNumber(num, row, col)) { //causes lots of errors
                        return false;
                    }
                }
            }
        }
        return true;
    }

    public int get(int x, int y) {
        return board[x][y];
    }

    public boolean set(int x, int y, int value) {
        if (value < 0 || value > 9) {
            return false;
        }

        try {

            if (value != 0) {
                if (!checkNumber(value, x, y)) {
                    return false;
                }
            }

            board[x][y] = value;
        } catch (ArrayIndexOutOfBoundsException e) {
            return false;
        }
        return true;
    }

    public String toString() {
        StringBuilder stringBuilder = new StringBuilder();

        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                if ((i == 3 || i == 6) && j == 0) {
                    stringBuilder.append("-----------------------------\n");
                }
                if ((j == 3 || j == 6)) {
                    stringBuilder.append("| ");
                }
                if (j == 8) {
                    stringBuilder.append(board[i][j]);
                    stringBuilder.append("\n");
                } else {
                    stringBuilder.append(board[i][j]);
                    stringBuilder.append("  ");
                }
            }
        }
        return stringBuilder.toString();
    }

    private boolean checkNumberInRow(int number, int row) {
        for (int c = 0; c < 9; c++) {
            if (board[row][c] == number) {
                return false;
            }
        }
        return true;
    }

    private boolean checkNumberInColumn(int number, int column) {
        for (int r = 0; r < 9; r++) {
            if (board[r][column] == number) {
                return false;
            }
        }
        return true;
    }

    private boolean checkNumberInBox(int number, int row, int column) {
        int tempR = row - row % 3;
        int tempC = column - column % 3;
        for (int r = tempR; r < tempR + 3; r++) {
            for (int c = tempC; c < tempC + 3; c++) {
                if (board[r][c] == number) {
                    return false;
                }
            }
        }
        return true;
    }

    private boolean checkNumber(int number, int row, int column) {

        return (checkNumberInRow(number, row)
                && checkNumberInColumn(number, column)
                && checkNumberInBox(number, row, column));
    }

}
